## 业务侧最小入口代码
```ts
import { Context } from '@kit.AbilityKit'
import { ColumnType, DatabaseConfig, OCORMInit, defineEntity } from 'ocorm'

let ready = false

export async function appReaderFirstStep(context: Context): Promise<void> {
  if (ready) {
    return
  }

  defineEntity('DocRouteUser', {
    tableName: 'doc_route_users',
    columns: [
      { property: 'id', name: 'id', primaryKey: true, autoIncrement: true },
      { property: 'name', name: 'name', type: ColumnType.TEXT }
    ]
  })

  await OCORMInit(context, {
    config: new DatabaseConfig('doc-route.db'),
    autoCreateTables: true
  })

  ready = true
}
```

## 验证步骤
1. 执行 `await appReaderFirstStep(this.context)`。
2. 无异常即证明你可以按“业务应用开发者阅读路径”继续。
3. 若异常包含“实体未注册/数据库未初始化”等关键词，先看 [03-10分钟最小闭环](./03-10分钟最小闭环.md)。
