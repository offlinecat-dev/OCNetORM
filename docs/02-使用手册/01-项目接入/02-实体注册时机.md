# 实体注册时机

## 何时用
项目启动阶段需要保证实体元数据可用时。

## 怎么用
严格顺序：`defineEntity` -> `OCORMInit` -> `Repository`。

```ts
import { ColumnType, DatabaseConfig, OCORMInit, defineEntity } from 'ocorm'

defineEntity('User', {
  tableName: 'users',
  columns: [
    { property: 'id', primaryKey: true, autoIncrement: true },
    { property: 'name', type: ColumnType.TEXT, nullable: false }
  ]
})

await OCORMInit(context, {
  config: new DatabaseConfig('app.db'),
  autoCreateTables: true
})
```

## 最小可运行模板
```ts
import { Context } from '@kit.AbilityKit'
import { ColumnType, DatabaseConfig, OCORMInit, Repository, defineEntity } from 'ocorm'

let initialized: boolean = false

export async function bootstrap(context: Context): Promise<void> {
  if (initialized) {
    return
  }

  defineEntity('Todo', {
    tableName: 'todos',
    columns: [
      { property: 'id', primaryKey: true, autoIncrement: true },
      { property: 'title', type: ColumnType.TEXT, nullable: false },
      { property: 'done', type: ColumnType.INTEGER, defaultValue: 0 }
    ]
  })

  await OCORMInit(context, {
    config: new DatabaseConfig('todo.db').setQueryTimeout(1200),
    autoCreateTables: true
  })

  initialized = true
}

export async function createTodo(title: string): Promise<void> {
  const repo = new Repository('Todo')
  await repo.save({ title, done: 0 })
}
```

## 失败示例与修复
失败示例（列先注册，实体未注册）：
```ts
import { Column, ColumnType } from 'ocorm'

class BrokenUser {}
const user = new BrokenUser()
Column({ name: 'name', type: ColumnType.TEXT })(user, 'name')
// 抛 EntityNotRegisteredError
```

修复：
```ts
import { Column, ColumnType, Table } from 'ocorm'

class FixedUser {}
const user = new FixedUser()
Table('users')(user.constructor as Function)
Column({ name: 'name', type: ColumnType.TEXT })(user, 'name')
```
